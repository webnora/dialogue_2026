```python
def get_top_confusions(cm, labels, top_n=3):
    """–í–æ–∑–≤—Ä–∞—â–∞–µ—Ç —Ç–æ–ø-N –ø—É—Ç–∞—é—â–∏—Ö—Å—è –∂–∞–Ω—Ä–æ–≤—ã—Ö –ø–∞—Ä"""
    confusions = []
    for i in range(len(labels)):
        for j in range(len(labels)):
            if i != j:
                confusions.append({
                    'true_genre': labels[i],
                    'predicted_genre': labels[j],
                    'count': cm[i, j],
                    'rate': cm[i, j] / cm[i].sum()  # –ø—Ä–æ—Ü–µ–Ω—Ç –æ—à–∏–±–æ–∫
                })

    return sorted(confusions, key=lambda x: -x['rate'])[:top_n]
```
